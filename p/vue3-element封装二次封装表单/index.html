<script lang="ts" setup>
  import {computed, useTemplateRef} from "vue";
  import {FormInstance} from "element-plus";

  const props = defineProps(['formItems', 'rules'])

  const modelValue = defineModel()

  const componentMap = {
    input: 'el-input',
    number: 'el-input-number',
    select: 'el-select',
  }

  const formInstance = useTemplateRef<FormInstance>('formRef')


  const whiteList = ['type', 'label', 'prop']

  const items = computed(() => props.formItems.filter(item => item.hidden !== true))

  function getProps(item: any) {
    if (item.props) return item.props
    return Object.keys(item).filter(key => !whiteList.includes(key)).reduce((prev, key) => {
      prev[key] = item[key]
      return prev
    }, {})
  }

  function getComponent(item: any) {
    const {type} = item
    if (type && typeof type !== "string") {
      return type
    }

    return componentMap[type || 'input']
  }

  defineExpose({
    validate(...args) {
      return formInstance.value?.validate(...args)
    }
  })
</script>

<template>
  <el-form ref="formRef" :model="modelValue" :rules="rules">
    <el-form-item v-for="item in items" :key="item.prop" :label="item.label" :prop="item.prop"
                  label-width="120" required>
      <slot :name="item.prop" :row="item">
        <component :is="getComponent(item)" v-model="modelValue[item.prop]" v-bind="getProps(item)">
          <template v-if="['select'].includes(item.type)">
            <el-option v-for="option in item.options" :key="option.value" :label="option.label" :value="option.value">
              <component v-if="option.slot !== 'string'" :is="option.slot"></component>
              <slot v-else :option="option" :name="option.slot">{{ option.label }}</slot>
            </el-option>
          </template>
        </component>
      </slot>
    </el-form-item>
  </el-form>
</template>
